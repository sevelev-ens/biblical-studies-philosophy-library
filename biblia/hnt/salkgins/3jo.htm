יוֹחָנָן ג

א    1 הַזָּקֵן אֶל-גָּיוֹס הַיָּקָר אֲשֶׁר אֲנִי אֹהֵב בְּתָם-לֵב׃ 2 כָּלְתָה נַפְשִׁי יְדִידִי כִּי תַצְלִיחַ בְּכָל-חֲפָצֶיךָ וְהָיִיתִי שָׁלֵם בַּגּוּף וּבַנֶּפֶשׁ וְאִישׁ מַצְלִיחַ׃ 3 כִּי שָׂמַחְתִּי מְאֹד כַּאֲשֶׁר בָּאוּ אַחִים וַיַּגִּידוּ עַל-אֲמִתֶּךָ כִּי בְדֶרֶךְ אֱמֶת אַתָּה הֹלֵךְ׃ 4 אֵין גְּדֹלָה מִשִּׂמְחָתִי כִּי אֶשְׁמַע עַל-בָּנַי אֲשֶׁר בְּדֶרֶךְ אֱמֶת יְהַלֵּכוּן׃ 5 בֶּאֱמוּנָה מַעֲשֶׂיךָ יְדִידִי כָּל-אֲשֶׁר אַתָּה עֹשֶׂה עִם-הָאַחִים הָאֹרְחִים׃ 6 אֲשֶׁר נָתְנוּ עֵדוּתָם עַל-אַהֲבָתְךָ בִּפְנֵי הַקָּהָל וְטוֹב תַּעֲשֶׂה כִּי תַעֲזֹר לָהֶם עַל-דַּרְכֵיהֶם לְרָצוֹן לִפְנֵי אֱלֹהִים׃ 7 כִּי לְמַעַן שְׁמוֹ יָצָאוּ וְלֹא לָקְחוּ מִיַּד הַגּוֹיִם מְאוּמָה׃ 8 עַל-כֵּן מוּטָל עָלֵינוּ לְקַבֵּל פְּנֵיהֶם וְלִהְיוֹת עֹזְרִים לָאֱמֶת׃ 9 וַאֲנִי כָתַבְתִּי אֶל-הָעֵדָה אַךְ דִּיּוֹטְרִיפַס הֶחָפֵץ לִהְיוֹת לְרֹאשׁ לָהֶם לֹא קִבֵּל אֹתָנוּ׃ 10 עַל-כֵּן בְּבֹאִי אֶפְקֹד עָלָיו מַעֲשָׂיו אֲשֶׁר עָשָׂה כִּי דִבֶּר סָרָה עָלֵינוּ וְאֵין דַּי לוֹ בָזֹאת כִּי אַף-לֹא יֹאבֶה לְקַבֵּל אֶת-הָאַחִים וְאֶת-אֵלֶּה אֲשֶׁר יֹאבוּ יַעְצָר-בָּם וּמִתּוֹךְ הַקָּהָל יְגָרְשֵׁם׃ 11 יְדִידִי אַל-תֵּלֵךְ בְּעִקְּבוֹת הָרָעִים כִּי אִם-בְּעִקְּבוֹת הַטּוֹבִים הָעֹשֶׂה טוֹב הוּא מֵאֵת אֱלֹהִים וְהַפֹּעֵל רָע אֶת-אֱלֹהִים לֹא רָאָה׃ 12 וְדִימֶטְרִיּוֹס יֶשׁ-לוֹ עֵדוּת מִכָּל-הָעֵדָה גַּם-מִפִּי הָאֱמֶת וְגַם-מִפִּינוּ וְיָדַעְתָּה כִּי עֵדוּתֵנוּ נֶאֱמָנָה׃ 13 רַבּוֹת לִי לִכְתֹּב אֵלֶיךָ אַךְ לֹא חָפַצְתִּי לִכְתֹּב בְּעֵט וּבִדְיוֹ׃ 14 אֲבָל אֲקַוֶּה לִרְאוֹתְךָ בִמְהֵרָה וּפֶה אֶל-פֶּה נְדַבֵּרָה שָׁלוֹם לָךְ רֵעֵינוּ פֹּקְדִים לִשְׁלוֹמְךָ וְאֶת-רֵעֵינוּ אִתְּךָ תִּפְקֹד לְשָׁלוֹם לְאִישׁ אִישׁ בִּשְׁמוֹ׃
