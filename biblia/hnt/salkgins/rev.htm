וַעֲבָדָיו יְשָׁרְתֻהוּ׃ 4 וְהֵם יִרְאוּ אֶת-פָּנָיו וְנָשְׂאוּ אֶת-שְׁמוֹ עַל-מִצְחָם׃ 5 וְאֵין לַיְלָה שָׁם וְאֵין צֹרֶךְ לָהֶם לְאוֹר נֵר וּלְאוֹר שָׁמֶשׁ כִּי-יְהוָֹה אֱלֹהִים יִתֵּן אוֹרוֹ עֲלֵיהֶם וּלְפָנָיו יִמְלְכוּ עַד-עוֹלְמֵי עוֹלָמִים׃ 6 וַיֹּאמֶר אֵלַי הַדְּבָרִים הָאֵלֶּה כֵּנִים וְנֶאֱמָנִים וַיהוָֹה אֱלֹהֵי רוּחוֹת הַנְּבִיאִים שָׁלַח אֶת-מַלְאָכוֹ לְהַרְאוֹת אֶת-עֲבָדָיו אֵת אֲשֶׁר תִּקְרֶינָה בְּקֶרֶב הַיָּמִים׃ 7 הִנְנִי בָא עַד-מְהֵרָה אַשְׁרֵי הַשֹּׁמֵר דִּבְרֵי הַנְּבוּאָה בַּסֵּפֶר הַזֶּה׃ 8 וַאֲנִי יוֹחָנָן שָׁמַעְתִּי וְרָאִיתִי אֶת-אֵלֶּה וַיְהִי בְּשָׁמְעִי וּבִרְאֹתִי אֶת-אֵלֶּה וָאֶפֹּל לְרַגְלֵי הַמַּלְאָךְ אֲשֶׁר-הֶרְאַנִי אֶת-אֵלֶּה לְהִשְׁתַּחֲוֹת לוֹ׃ 9 וַיֹּאמֶר אֵלַי לֹא-כֵן רְאֵה אָנֹכִי עֶבֶד כָּמוֹךָ וְחָבֵר אֲנִי לְךָ וּלְאַחֶיךָ הַנְּבִיאִים וְלַשֹּׁמְרִים אֶת-דִּבְרֵי הַסֵּפֶר הַזֶּה לֵאלֹהִים תִּשְׁתַּחֲוֶה׃ 10 וַיֹּאמֶר אֵלַי עוֹד אַל-תַּחְתֹּם אֶת-דִּבְרֵי הַנְּבוּאָה אֲשֶׁר בַּסֵּפֶר הַזֶּה כִּי קָרוֹב הַמּוֹעֵד׃ 11 בֶּן-עַוְלָה יוֹסִיף עֲשׂוֹת עָוֶל וְנִתְעָב יוֹסִיף עַל-תּוֹעֲבָתוֹ אִישׁ צַדִּיק יוֹסִיף עַל-צִדְקָתוֹ וְקָדוֹשׁ יוֹסִיף לְהִתְקַדֵּשׁ בִּקְדֻשָּׁתוֹ׃ 12 וַאֲנִי הִנְנִי בָא עַד-מְהֵרָה וּשְׂכָרִי אִתִּי לְשַׁלֵּם לְאִישׁ אִישׁ כְּמַעֲשֵׂהוּ׃ 13 אָנֹכִי אָלֶף וְתָו רִאשׁוֹן וְאַחֲרוֹן רֹאשׁ וָסוֹף׃ 14 אַשְׁרֵי הַשּׁוֹמְרִים אֶת-מִצְוֹתָיו וְהָיָה לָהֶם רִשְׁיוֹן לֶאֱכֹל מֵעֵץ הַחַיִּים וְלָבֹא הָעִירָה דֶּרֶךְ שְׁעָרֶיהָ׃ 15 וּמִחוּץ לָעִיר הַכְּלָבִים וְהַקֹּסְמִים וְהַזֹּנִים וְהַמְרַצְּחִים וְעֹבְדֵי הָאֱלִילִים וְכָל-אֹהֵב כָּזָב וּפֹעֵל שָׁקֶר׃ 16 אֲנִי יֵשׁוּעַ שָׁלַחְתִּי אֶת-מַלְאָכִי לְהָעִיד לָכֶם עַל-אֵלֶּה לְעֵינֵי הַקְּהִלּוֹת אָנֹכִי שֹׁרֶשׁ דָּוִד וְחֹטֶר מִמֶּנּוּ אוֹר נֹגַהּ וְכוֹכַב הַשָּׁחַר׃ 17 וְהָרוּחַ וְהַכַּלָּה אֹמְרִים בֹּאָה-נָּא וְהַשֹּׁמֵעַ יֹאמַר בֹּאָה-נָּא וְכָל-צָמֵא יָבֹא וְכָל-הֶחָפֵץ יִקַּח מַיִם חַיִּים חִנָּם׃ 18 הַעִדֹתִי בְכָל-הַשֹּׁמֵעַ דִּבְרֵי נְבוּאַת הַסֵּפֶר הַזֶּה אִם-יוֹסִיף אִישׁ עֲלֵיהֶם יוֹסִיף אֱלֹהִים עָלָיו אֶת-הַמַּכּוֹת הַכְּתוּבוֹת בַּסֵּפֶר הַזֶּה׃ 19 וְאִם-יְגְרַע אִישׁ מִדִּבְרֵי סֵפֶר הַנְּבוּאָה הַזֹּאת יִגְרַע אֱלֹהִים אֶת-חֶלְקוֹ מֵסֵפֶר הַחַיִּים וּמֵעִיר הַקֹּדֶשׁ הַכְּתֻבִים בַּסֵּפֶר הַזֶּה׃ 20 וְהַמֵּעִיד דְּבָרִים אֵלֶּה עָנָה לֵאמֹר אָכֵן אֲנִי בָא עַד-מְהֵרָה אָמֵן בֹּאָה-נָּא הָאָדוֹן יֵשׁוּעַ׃ 21 חֶסֶד יֵשׁוּעַ הַמָּשִׁיחַ אֲדֹנֵינוּ עִם-כֻּלְכֶם אָמֵן׃
